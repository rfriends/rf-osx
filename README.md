rfriends_osxはosx環境でrfriends3を動作させるスクリプトです  
  
cd ~/  
wget -O rfriends_osx-main.zip https://github.com/rfriends/rfriends_osx/archive/master.zip  
unzip rfriends_osx-main.zip  
cd rfriends_osx-main  
sh rfriends3_osx.sh
